        // ---------- TRIEDA CHIBI FIGHTER (vizuál podľa návrhu) ----------
        class ChibiFighter {
            constructor(id, name, heroType, isBot, startX, startZ) {
                this.id = id;
                this.name = name;
                this.heroType = heroType;
                this.isBot = isBot;
                this.health = 100;
                this.maxHealth = 100;
                this.position = { x: startX, z: startZ };
                this.mesh = null;
                this.label = null;
                this.speed = 4.6;
                this.attackCooldown = 0;
                this.ammo = 30;
                this.reserveAmmo = 90;
                this.skillCooldowns = [0, 0, 0];
            }
            
            createModel() {
                this.mesh = new THREE.Group();
                this.mesh.position.set(this.position.x, 0, this.position.z);
                scene.add(this.mesh);

                const bodyGroup = new THREE.Group();
                this.mesh.add(bodyGroup);

                // Attempt to load a GLB model from `models/<HERO>.glb`. If loading fails, fall back to procedurally generated primitive.
                try {
                    const loader = new GLTFLoader();
                    const fbxLoader = new FBXLoader();
                    // If a specific model file was assigned to this fighter, try it first
                    if (this.modelFile) {
                        const mf = this.modelFile;
                        if (mf.toLowerCase().endsWith('.glb')) {
                            loader.load(mf, (gltf) => {
                                while (bodyGroup.children.length) bodyGroup.remove(bodyGroup.children[0]);
                                const modelRoot = gltf.scene;
                                modelRoot.traverse((n) => { if (n.isMesh) { n.castShadow = true; n.receiveShadow = true; } });
                                modelRoot.position.set(0, 0, 0);
                                bodyGroup.add(modelRoot);
                            }, undefined, () => { /* falled back to search below */ });
                        } else if (mf.toLowerCase().endsWith('.fbx')) {
                            fbxLoader.load(mf, (obj) => {
                                while (bodyGroup.children.length) bodyGroup.remove(bodyGroup.children[0]);
                                obj.traverse((n) => { if (n.isMesh) { n.castShadow = true; n.receiveShadow = true; } });
                                obj.position.set(0, 0, 0);
                                bodyGroup.add(obj);
                            }, undefined, () => { /* falled back to search below */ });
                        }
                    }

                    const candidates = [
                        `my-html-game/models/${this.heroType}.glb`,
                        `models/${this.heroType}.glb`,
                        `${this.heroType}.glb`,
                    ];
                    const fbxCandidates = [
                        `my-html-game/models/${this.heroType}.fbx`,
                        `models/${this.heroType}.fbx`,
                        `${this.heroType}.fbx`,
                    ];

                    let loaded = false;
                    const tryNextGLB = (index) => {
                        if (index >= candidates.length) { tryNextFBX(0); return; }
                        const path = candidates[index];
                        loader.load(path, (gltf) => {
                            while (bodyGroup.children.length) bodyGroup.remove(bodyGroup.children[0]);
                            const modelRoot = gltf.scene;
                            modelRoot.traverse((n) => { if (n.isMesh) { n.castShadow = true; n.receiveShadow = true; } });
                            modelRoot.position.set(0, 0, 0);
                            bodyGroup.add(modelRoot);
                            loaded = true;
                        }, undefined, () => { tryNextGLB(index+1); });
                    };

                    const tryNextFBX = (index) => {
                        if (index >= fbxCandidates.length) { console.warn('No model found for', this.heroType); return; }
                        const path = fbxCandidates[index];
                        fbxLoader.load(path, (obj) => {
                            while (bodyGroup.children.length) bodyGroup.remove(bodyGroup.children[0]);
                            obj.traverse((n) => { if (n.isMesh) { n.castShadow = true; n.receiveShadow = true; } });
                            obj.position.set(0, 0, 0);
                            bodyGroup.add(obj);
                            loaded = true;
                        }, undefined, () => { tryNextFBX(index+1); });
                    };

                    tryNextGLB(0);
                } catch (e) {
                    console.warn('GLTFLoader not available or load error', e);
                    return;
                }
                
                const div = document.createElement('div');
                div.textContent = `${this.name} ❤️ ${Math.floor(this.health)}`;
                div.style.background = 'rgba(0,0,0,0.8)';
                div.style.color = '#ffdd99';
                div.style.padding = '3px 10px';
                div.style.borderRadius = '20px';
                div.style.fontWeight = 'bold';
                div.style.fontSize = '11px';
                div.style.border = `1px solid ${new THREE.Color(HEROES[this.heroType].color).getStyle()}`;
                this.label = new CSS2DObject(div);
                this.label.position.set(0, 1.45, 0);
                this.mesh.add(this.label);
            }
            
            updateLabel() {
                if (this.label && this.label.element) this.label.element.textContent = `${this.name} ❤️ ${Math.max(0, Math.floor(this.health))}`;
            }
            setPosition(x, z) {
                const bound = 26;
                this.position.x = Math.min(bound, Math.max(-bound, x));
                this.position.z = Math.min(bound, Math.max(-bound, z));
                if (this.mesh) this.mesh.position.set(this.position.x, 0, this.position.z);
            }
            damage(amount, sourceId) {
                this.health = Math.max(0, this.health - amount);
                this.updateLabel();
                if (this.health <= 0) {
                    addKillMessage(`${this.name} bol porazený!`);
                    if (sourceId !== undefined && sourceId !== this.id) {
                        const killer = fighters.find(f => f.id === sourceId);
                        if (killer && !killer.isBot && killer.id === localPlayer?.id) {
                            if (killer.ammo + 12 <= 30) killer.ammo += 12; else killer.ammo = 30;
                            updateAmmoUI();
                        }
                    }
                    this.respawn();
                }
                updateLobbyCount();
            }
            respawn() {
                this.health = this.maxHealth;
                this.ammo = 30;
                this.reserveAmmo = 90;
                const randX = (Math.random() - 0.5) * 44;
                const randZ = (Math.random() - 0.5) * 44;
                this.setPosition(randX, randZ);
                this.updateLabel();
                if (!this.isBot && this.id === localPlayer?.id) updateAmmoUI();
            }
            shoot() {
                if (this.attackCooldown > 0) return false;
                if (this.ammo <= 0) {
                    if (this.reserveAmmo > 0) {
                        let needed = Math.min(30 - this.ammo, this.reserveAmmo);
                        this.ammo += needed;
                        this.reserveAmmo -= needed;
                        if (!this.isBot && this.id === localPlayer?.id) { updateAmmoUI(); addKillMessage("🔁 Prebitie!"); }
                    }
                    return false;
                }
                this.attackCooldown = HEROES[this.heroType].fireRate;
                this.ammo--;
                if (!this.isBot && this.id === localPlayer?.id) updateAmmoUI();
                return true;
            }
        }

        function addKillMessage(msg) {
            const feed = document.getElementById('killFeed');

        // Map builder (herná aréna)
        function buildMap(mapId) {
            const toRemove = [];
            scene.children.forEach(child => { if (child.userData && child.userData.isMapPart) toRemove.push(child); });
            toRemove.forEach(item => scene.remove(item));
            const mapData = MAPS[mapId];
            scene.background = new THREE.Color(mapData.bgColor);
            const groundMat = new THREE.MeshStandardMaterial({ color: mapData.groundColor });
            const groundPlane = new THREE.Mesh(new THREE.PlaneGeometry(60, 60), groundMat);
            groundPlane.rotation.x = -Math.PI/2;
            groundPlane.position.y = -0.7;
            groundPlane.receiveShadow = true;
            groundPlane.userData = { isMapPart: true };
            scene.add(groundPlane);
            const ringMat = new THREE.MeshStandardMaterial({ color: 0xccaa77 });
            const ring = new THREE.Mesh(new THREE.TorusGeometry(27, 0.45, 32, 120), ringMat);
            ring.rotation.x = Math.PI/2;
            ring.position.y = -0.55;
            ring.userData = { isMapPart: true };
            scene.add(ring);
            // Dekorácie
            if (mapData.decoType === "crystal") {
                for (let i=0;i<90;i++) { const c=new THREE.Mesh(new THREE.ConeGeometry(0.45,1.4,6), new THREE.MeshStandardMaterial({ color:0xaa88ff,emissive:0x4422aa })); c.position.set((Math.random()-0.5)*52,-0.5,(Math.random()-0.5)*52); c.castShadow=true; c.userData={isMapPart:true}; scene.add(c); }
            } else if (mapData.decoType === "trees") {
                for (let i=0;i<70;i++) { const t=new THREE.Mesh(new THREE.CylinderGeometry(0.4,0.6,1.2,6), new THREE.MeshStandardMaterial({ color:0x8b5a2b })); t.position.set((Math.random()-0.5)*50,-0.5,(Math.random()-0.5)*50); t.castShadow=true; t.userData={isMapPart:true}; scene.add(t); }
            } else if (mapData.decoType === "lava") {
                for (let i=0;i<60;i++) { const r=new THREE.Mesh(new THREE.DodecahedronGeometry(0.55), new THREE.MeshStandardMaterial({ color:0xff5522,emissive:0x552200 })); r.position.set((Math.random()-0.5)*52,-0.55,(Math.random()-0.5)*52); r.castShadow=true; r.userData={isMapPart:true}; scene.add(r); }
            }
            const particleCount=1000, geo=new THREE.BufferGeometry(), pos=new Float32Array(particleCount*3);
            for(let i=0;i<particleCount;i++) { pos[i*3]=(Math.random()-0.5)*110; pos[i*3+1]=Math.random()*16; pos[i*3+2]=(Math.random()-0.5)*100; }
            geo.setAttribute('position', new THREE.BufferAttribute(pos,3));
            const stars=new THREE.Points(geo, new THREE.PointsMaterial({ color:0xffcc88, size:0.07 }));
            stars.userData={isMapPart:true};
            scene.add(stars);
        }
        
        // Čistenie celej hry (odstránenie všetkých bojovníkov)

            if (fighters.length) {
                fighters.forEach(fighter => {
                    if (fighter.mesh) scene.remove(fighter.mesh);
                    if (fighter.label) scene.remove(fighter.label);

            }

            scene.children.forEach(child => { if (child.userData && child.userData.isMapPart) toRemove.push(child); });
            toRemove.forEach(item => scene.remove(item));
        }
        
        // Inicializácia hráčov (podľa vybranej postavy pre lokálneho)
        function initFighters(selectedHeroId) {
            fighters = [];
            const heroOptions = ["LUMA", "NOKI", "PIXIE"];
            for (let i=0;i<MAX_PLAYERS;i++) {
                const isBot = i !== 0;
                let heroPick;
                if (!isBot) heroPick = selectedHeroId;
                else heroPick = heroOptions[i % heroOptions.length];

                const startX = (Math.random() - 0.5) * 40;
                const startZ = (Math.random() - 0.5) * 40;
                const fighter = new ChibiFighter(i, playerName, heroPick, isBot, startX, startZ);
                // Assign a user model (rotate through the list) so we use your uploaded characters
                const modelIndex = i % USER_MODELS.length;
                fighter.modelFile = USER_MODELS[modelIndex];
                fighters.push(fighter);
                fighter.createModel();

        // Zvyšné herné mechaniky (raycast, schopnosti, AI)
        function raycastHitEnemy(origin, direction) {
            let closest=null, minD=999;
            for(let f of fighters) {
                if(f===localPlayer||f.health<=0) continue;
                const toF=new THREE.Vector3(f.position.x,0.8,f.position.z).sub(origin);
                const d=toF.length();
                const dot=direction.clone().normalize().dot(toF.clone().normalize());
                if(dot>0.85 && d<3.7 && d<minD) { minD=d; closest=f; }
            }
            return closest;
        }
        function createMuzzleFlash(pos,dir){ const f=new THREE.PointLight(0xffaa66,1.2,6); f.position.copy(pos.clone().add(dir.clone().multiplyScalar(0.6))); scene.add(f); setTimeout(()=>scene.remove(f),70); }
        function getClosestEnemy(){ if(!localPlayer)return null; let c=null,minD=Infinity; for(let f of fighters) if(f!==localPlayer&&f.health>0){ const d=Math.hypot(f.position.x-localPlayer.position.x, f.position.z-localPlayer.position.z); if(d<minD){minD=d;c=f;} } return c; }
        function areaDamage(dmg,source){ for(let f of fighters) if(f!==source&&f.health>0&&Math.hypot(f.position.x-source.position.x,f.position.z-source.position.z)<6) f.damage(dmg,source.id); }
        function createBeamEffect(s,t){ if(t&&t.mesh){ const pts=[s.mesh.position.clone(), t.mesh.position.clone()]; const geo=new THREE.BufferGeometry().setFromPoints(pts); const line=new THREE.Line(geo,new THREE.LineBasicMaterial({color:0xffaa66})); scene.add(line); setTimeout(()=>scene.remove(line),180); } }
        function createShieldEffect(f){ const s=new THREE.Mesh(new THREE.SphereGeometry(0.95,16,16),new THREE.MeshStandardMaterial({color:0xffcc88,emissive:0xff8844,transparent:true})); s.position.copy(f.mesh.position); scene.add(s); setTimeout(()=>scene.remove(s),1000); }
        function useHeroSkill(idx){
            if(!localPlayer||localPlayer.skillCooldowns[idx]>0) return;
            localPlayer.skillCooldowns[idx]=7.5;
            const hero=localPlayer.heroType, target=getClosestEnemy();
            if(idx===0){
                if(hero==="LUMA"){ if(target) target.damage(38,localPlayer.id); createBeamEffect(localPlayer,target); addKillMessage(`✨ ${HEROES.LUMA.skill1.name}!`);}
                else if(hero==="NOKI"){ if(target){ target.damage(28,localPlayer.id); target.speed*=0.45; setTimeout(()=>{if(target)target.speed=4.6;},2000);} addKillMessage(`🌿 ${HEROES.NOKI.skill1.name}!`);}
                else if(hero==="PIXIE"){ areaDamage(24,localPlayer); addKillMessage(`💣 ${HEROES.PIXIE.skill1.name}!`);}
            } else if(idx===1){
                if(hero==="LUMA"){ localPlayer.health=Math.min(localPlayer.maxHealth,localPlayer.health+28); localPlayer.updateLabel(); createShieldEffect(localPlayer); addKillMessage(`🛡️ ${HEROES.LUMA.skill2.name}!`);}
                else if(hero==="NOKI"){ localPlayer.health=Math.min(localPlayer.maxHealth,localPlayer.health+22); addKillMessage(`💚 ${HEROES.NOKI.skill2.name}!`);}
                else if(hero==="PIXIE"){ localPlayer.health+=18; localPlayer.reserveAmmo+=15; if(localPlayer.reserveAmmo>180)localPlayer.reserveAmmo=180; addKillMessage(`🤖 ${HEROES.PIXIE.skill2.name}!`);}
                updateAmmoUI();
            } else if(idx===2){
                if(hero==="LUMA"){ areaDamage(42,localPlayer); addKillMessage(`⭐ ${HEROES.LUMA.skill3.name}!`);}
                else if(hero==="NOKI"){ areaDamage(38,localPlayer); addKillMessage(`🍃 ${HEROES.NOKI.skill3.name}!`);}
                else if(hero==="PIXIE"){ if(target) target.damage(50,localPlayer.id); else areaDamage(30,localPlayer); addKillMessage(`⚡ ${HEROES.PIXIE.skill3.name}!`);}
            }
            updateLobbyCount(); updateAmmoUI();
        }
        
        function updateMovementAndAI(delta){
            if(!gameActive) return;
            if(localPlayer){
                let mx=0,mz=0;
                if(keyStates.w) mz-=1; if(keyStates.s) mz+=1; if(keyStates.a) mx-=1; if(keyStates.d) mx+=1;
                if(mx!==0||mz!==0){
                    const len=Math.hypot(mx,mz); mx/=len; mz/=len;
                    const forward={x:Math.sin(yawAngle),z:Math.cos(yawAngle)};
                    const right={x:Math.cos(yawAngle),z:-Math.sin(yawAngle)};
                    let newX=localPlayer.position.x+(forward.x*mz+right.x*mx)*localPlayer.speed*delta;
                    let newZ=localPlayer.position.z+(forward.z*mz+right.z*mx)*localPlayer.speed*delta;
                    localPlayer.setPosition(newX,newZ);
                }
                camera.position.set(localPlayer.position.x,1.68,localPlayer.position.z);
                camera.rotation.order='YXZ'; camera.rotation.set(pitchAngle,yawAngle,0);
            }
            for(let bot of fighters){
                if(bot.isBot&&bot.health>0){
                    let closest=null,minD=Infinity;
                    for(let other of fighters) if(other!==bot&&other.health>0){ const d=Math.hypot(other.position.x-bot.position.x,other.position.z-bot.position.z); if(d<minD){minD=d;closest=other;} }
                    if(closest){
                        const dx=closest.position.x-bot.position.x, dz=closest.position.z-bot.position.z, len=Math.hypot(dx,dz);
                        if(len>1.2){ const move=Math.min(bot.speed*delta,len-0.8); bot.setPosition(bot.position.x+(dx/len)*move,bot.position.z+(dz/len)*move); }
                        else if(len<=2.8&&bot.attackCooldown<=0&&bot.ammo>0){ bot.ammo--; closest.damage(HEROES[bot.heroType].damage,bot.id); bot.attackCooldown=HEROES[bot.heroType].fireRate; createMuzzleFlash(bot.mesh.position,new THREE.Vector3(dx,0,dz).normalize()); }
                        else if(bot.ammo<=0&&bot.reserveAmmo>0){ let load=Math.min(30,bot.reserveAmmo); bot.ammo+=load; bot.reserveAmmo-=load; }
                    }
                }
                if(bot.attackCooldown>0) bot.attackCooldown-=delta;
                for(let i=0;i<3;i++) if(bot.skillCooldowns[i]>0) bot.skillCooldowns[i]-=delta;
            }
        }
